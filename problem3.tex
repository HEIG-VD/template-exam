% Set length of fillin
\setlength{\fillinlinelength}{5cm}

Considérez le circuit électronique suivant :

\begin{figure}[h]
    \begin{center}
        \begin{circuitikz}
            \draw
            (0,0) to [short, *-] (6,0)
            to [V, l_=$\mathrm{j}{\omega}_m \underline{\phi}^s_R$] (6,2)
            to [R, l_=$R_R$] (6,4)
            to [short, i_=$\underline{i}^s_R$] (5,4)
            (0,0) to [open, v^>=$\underline{u}^s_s$] (0,4)
            to [short, *- ,i=$\underline{i}^s_s$] (1,4)
            to [R, l=$R_s$] (3,4)
            to [L, l=$L_{\sigma}$] (5,4)
            to [short, i_=$\underline{i}^s_M$] (5,3)
            to [L, l_=$L_M$] (5,0);
        \end{circuitikz}
        \caption{Circuit électrique \label{circuit}}
    \end{center}
\end{figure}

Connaissant les éléments suivants :\par
\vskip 1em
\begin{itemize}
    \item \( \underline{\phi}^s_R = 230 \angle 0^{\circ} \, \si{\volt} \)
    \item \( R_R = 0.1 \, \si{\ohm} \)
    \item \( R_s = 0.2 \, \si{\ohm} \)
    \item \( L_{\sigma} = 0.1 \, \si{\milli\henry} \)
    \item \( L_M = 0.2 \, \si{\milli\henry} \)
\end{itemize}
\vfil

Calculez les grandeurs suivantes :

\begin{parts}
    \part La réactance de \(L_{\sigma}\) à la fréquence angulaire \(\omega_m\).
    \fillin[$j\omega_m L_{\sigma}$]

    \part La réactance de \(L_M\) à la fréquence angulaire \(\omega_m\).
    \fillin[$j\omega_m L_M$]

    \part La tension \(\underline{u}^s_s\) en fonction de \( \underline{\phi}^s_R \), \( R_R \), et \( L_{\sigma} \).
    \fillin[$\underline{u}^s_s = \underline{\phi}^s_R - \underline{i}^s_R (R_R + j\omega_m L_{\sigma})$]

    \part Le courant \(\underline{i}^s_R\) à partir de \(\underline{\phi}^s_R\), \( R_R \), et \( L_{\sigma} \).
    \fillin[$\underline{i}^s_R = \frac{\underline{\phi}^s_R}{R_R + j\omega_m L_{\sigma}}$]

    \part La tension de sortie \(\underline{u}^s_s\) si \(\omega_m = 2\pi \times 50 \, \si{\hertz}\).
    \fillin[$\underline{u}^s_s = \underline{\phi}^s_R - \underline{i}^s_R (R_R + j2\pi \times 50 \times 0.1 \times 10^{-3})$]
\end{parts}
